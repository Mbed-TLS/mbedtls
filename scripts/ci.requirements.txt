# Python package requirements for Mbed TLS testing.

-r driver.requirements.txt

# Use a known version of Pylint, because new versions tend to add warnings
# that could start rejecting our code.
# 2.13.9 is the last version that supports Python 3.6.
pylint == 2.13.9

# mypy 0.971 is the last version that supports Python 3.6.
mypy >= 0.971

# tests/scripts/audit-validity-dates.py needs cryptography >= 35.0.0.
# To reduce future version churn, require the last version that supports
# Python 3.6.
# Only require cryptography on systems where pre-built wheels are
# available (at least for x86_64), because building locally requires
# a Rust compiler and we want to avoid that dependency.
cryptography >= 40.0.2; platform_system == "Linux" or platform_system == "Windows" or platform_system == "Darwin"
types-cryptography

# For building `tests/data_files/server9-bad-saltlen.crt` and check python
# files.
asn1crypto
